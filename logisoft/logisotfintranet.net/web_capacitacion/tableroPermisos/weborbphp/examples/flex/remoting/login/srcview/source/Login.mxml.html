<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" layout="absolute" 
    height="221" viewSourceURL="srcview/index.html" width="446"
    creationComplete="controller.init( model, this )">
    <mx:states>
        <mx:State name="loggedin">
            <mx:SetProperty target="{button1}" name="label" value="Logout"/>
            <mx:SetEventHandler target="{button1}" name="click" handler="doLogout()"/>
            <mx:SetProperty target="{button1}" name="x" value="321"/>
        </mx:State>
    </mx:states>
    <mx:Script>
        <![CDATA[
            import mx.managers.PopUpManager;
            private var controller:Controller = new Controller();            
            
            [Bindable]
            private var model:Model = new Model();
            
            private function doLogin():void
            {
                var popup:LoginBox = PopUpManager.createPopUp( this, LoginBox, true ) as LoginBox;
                popup.setController( controller );
                PopUpManager.centerPopUp( popup );
            }
            
            private function doLogout():void
            {
                controller.logout();
            }
        ]]>
    </mx:Script>
    <mx:Canvas horizontalCenter="0" verticalCenter="0">
        <mx:DataGrid x="0" y="30" width="387" height="120" dataProvider="{model.products}">
            <mx:columns>
                <mx:DataGridColumn dataField="NAME"/>
                <mx:DataGridColumn dataField="PRICE"/>
            </mx:columns>
        </mx:DataGrid>
        <mx:Button x="0" y="0" label="Load Products (secure invocation)" click="controller.getProducts()"/>
        <mx:Button x="331" y="0" label="Login" click="doLogin()" id="button1"/>
    </mx:Canvas>
    
</mx:Application>
