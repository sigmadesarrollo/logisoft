    <mx:Script>
        <![CDATA[
            import mx.managers.PopUpManager;
            import mx.core.Application;
            import mx.controls.Alert;
            private function setCredentials():void
            {
                if( username.text.length == 0 || password.text.length == 0 )
                {
                    Alert.show( "Both username and password are required", "Incomplete Form" );
                    return;
                }
                
                Application.application.setCredentials( username.text, password.text );
                PopUpManager.removePopUp( this );
            }
        ]]>
    </mx:Script>

    <mx:Text x="93" y="53" text="hint: password is password" fontSize="9"/>
    <mx:Button x="109" y="106" label="Login" click="setCredentials()"/>
