require_once("PhoneBookRecord.php");

class PhoneBook
{
    var $records;

	public function __construct()
	{
		$item = new PhoneBookRecord();
		$item->firstName = "Midnight";
		$item->lastName = "Coders";
		$item->phoneNumber = "972.342.6908";			

		$this->records[] = $item;		
	}    
    
    public function addEntry(PhoneBookRecord $record)
    {
        $this->records[]=$record;
        return $record;
    }

    public function getEntries()
    {
        return $this->records;
    }
}
