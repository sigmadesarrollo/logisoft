package FoodAndDrinks.Codegen
     {
      
        import FoodAndDrinks.AccountDataMapper;
      
        import FoodAndDrinks.OrderDataMapper;
      
        import FoodAndDrinks.OrderLineDataMapper;
      
        import FoodAndDrinks.ProductDataMapper;
      
       public class _DataMapperRegistry
       {
        

          private var m_accountDataMapper:AccountDataMapper;

          public function get Account():AccountDataMapper
          {
            if(m_accountDataMapper == null )
              m_accountDataMapper = new AccountDataMapper();
              
            return m_accountDataMapper;
          }
        

          private var m_orderDataMapper:OrderDataMapper;

          public function get Order():OrderDataMapper
          {
            if(m_orderDataMapper == null )
              m_orderDataMapper = new OrderDataMapper();
              
            return m_orderDataMapper;
          }
        

          private var m_orderLineDataMapper:OrderLineDataMapper;

          public function get OrderLine():OrderLineDataMapper
          {
            if(m_orderLineDataMapper == null )
              m_orderLineDataMapper = new OrderLineDataMapper();
              
            return m_orderLineDataMapper;
          }
        

          private var m_productDataMapper:ProductDataMapper;

          public function get Product():ProductDataMapper
          {
            if(m_productDataMapper == null )
              m_productDataMapper = new ProductDataMapper();
              
            return m_productDataMapper;
          }
            
        }
      }
