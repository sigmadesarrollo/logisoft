package UI.Login
{
    import flash.events.Event;
    import FoodAndDrinks.Account;

    public class LoginEvent extends Event
    {
        private var _account:Account;
        
        public function LoginEvent(account:Account)
        {
            super("login");
                        
            _account = account;
        }
        
        public function get account():Account
        {
            return _account;
        }

    }
}
